function getRequest(type) {
var url,
theRequest = {},
i, strs;
type = type || 'search';
switch (type) {
case 'hash':
url = window.location.href.split('#')[1] || '';
break;
case 'search':
// 避免hash不规范情况
url = window.location.href.split('?')[1] || '';
url = url.split('#')[0];
break;
}
strs = url.split('&');
for (i = 0; i < strs.length; i++) {
theRequest[strs[i].split('=')[0]] = unescape(strs[i].split('=')[1]);
}
return theRequest;
}
